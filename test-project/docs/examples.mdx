# More Examples

## Web Development Languages

<LanguageTabs>
  <LanguageTab language="javascript">
    ```js
    // Fetch data from an API
    async function fetchData() {
      const response = await fetch('https://api.example.com/data');
      const data = await response.json();
      return data;
    }
    ```
  </LanguageTab>
  <LanguageTab language="typescript">
    ```ts
    // Fetch data from an API with types
    interface ApiResponse {
      id: number;
      name: string;
    }
    
    async function fetchData(): Promise<ApiResponse[]> {
      const response = await fetch('https://api.example.com/data');
      const data: ApiResponse[] = await response.json();
      return data;
    }
    ```
  </LanguageTab>
  <LanguageTab language="bun">
    ```ts
    // Bun native HTTP server
    const server = Bun.serve({
      port: 3000,
      fetch(request) {
        return new Response("Hello from Bun!");
      },
    });
    
    console.log(`Listening on http://localhost:${server.port}`);
    ```
  </LanguageTab>
  <LanguageTab language="php">
    ```php
    <?php
    // Fetch data from an API
    $ch = curl_init('https://api.example.com/data');
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    $response = curl_exec($ch);
    $data = json_decode($response, true);
    curl_close($ch);
    ?>
    ```
  </LanguageTab>
</LanguageTabs>

## System Programming

<LanguageTabs>
  <LanguageTab language="rust">
    ```rust
    fn main() {
        let numbers = vec![1, 2, 3, 4, 5];
        let sum: i32 = numbers.iter().sum();
        println!("Sum: {}", sum);
    }
    ```
  </LanguageTab>
  <LanguageTab language="c">
    ```c
    #include <stdio.h>
    
    int main() {
        int numbers[] = {1, 2, 3, 4, 5};
        int sum = 0;
        for (int i = 0; i < 5; i++) {
            sum += numbers[i];
        }
        printf("Sum: %d\n", sum);
        return 0;
    }
    ```
  </LanguageTab>
  <LanguageTab language="cpp">
    ```cpp
    #include <iostream>
    #include <vector>
    #include <numeric>
    
    int main() {
        std::vector<int> numbers = {1, 2, 3, 4, 5};
        int sum = std::accumulate(numbers.begin(), numbers.end(), 0);
        std::cout << "Sum: " << sum << std::endl;
        return 0;
    }
    ```
  </LanguageTab>
</LanguageTabs>
